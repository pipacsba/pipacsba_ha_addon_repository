  #include <abstractions/nameservice>

  capability setgid,
  capability setuid,
  capability dac_override,
  capability dac_read_search,
  capability net_bind_service,
  capability chown,

  # S6-Overlay v3 core execution (Fixes the /etc/s6-overlay error)
  /init rix,

  /run.sh rix,

  /etc/nginx/** r,

  /var/log/nginx/ r,
  /var/log/nginx/** rwk,

  # Logging and System
  /dev/stdout rw,

  signal (send, receive) peer=frigate_ingress_proxy,

  ptrace (trace) peer=unconfined,
}

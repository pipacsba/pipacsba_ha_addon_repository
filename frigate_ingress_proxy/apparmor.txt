#include <tunables/global>

# Profile name MUST match your slug: frigate_ingress_proxy
profile frigate_ingress_proxy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities required for S6-Overlay v3 and Nginx
  capability setgid,
  capability setuid,
  capability dac_override,
  capability dac_read_search,
  capability net_bind_service,
  capability chown,
  capability fowner,
  capability kill,
  capability sys_ptrace,

  # Network access
  network inet stream,
  network inet6 stream,
  network unix stream,

  # Core S6 and System binaries
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /sbin/nginx rix,
  /usr/sbin/nginx rix,
  /package/** rix,
  /command/** rix,
  /etc/s6-overlay/** r,

  # Crucial for S6 v3 execution from /run
  /run/ r,
  /run/** rwkix,

  # Addon scripts and configuration
  /run.sh rix,
  /etc/nginx/** r,
  /tmp/ r,
  /tmp/** rwk,
  /tmp/nginx.conf r,
  
  # Nginx internal paths (Fixes the Permission Denied on error.log)
  /var/lib/nginx/ r,
  /var/lib/nginx/** rwk,
  /var/log/nginx/ r,
  /var/log/nginx/** rwk,

  # Logging and System
  /dev/stdout rw,
  /dev/stderr rw,
  /dev/tty rw,
  /dev/null rw,
  /dev/random r,
  /dev/urandom r,

  # Bashio and Home Assistant internal access
  /usr/lib/bashio/** r,
  /data/options.json r,

  # Signals and process management
  signal (send, receive) peer=frigate_ingress_proxy,
  signal (receive) peer=unconfined,
  ptrace (trace) peer=unconfined,
}

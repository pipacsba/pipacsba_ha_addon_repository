#include <tunables/global>

# Profile name must match the slug: frigate_ingress_proxy
profile frigate_ingress_proxy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities required for S6-Overlay v3 and Nginx
  capability setgid,
  capability setuid,
  capability dac_override,
  capability dac_read_search,
  capability net_bind_service,
  capability chown,
  capability fowner,
  capability kill,

  # Network access
  network inet stream,
  network inet6 stream,

  # Core S6 and System binaries
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /sbin/nginx rix,
  /usr/sbin/nginx rix,
  /package/** rix,
  /command/** rix,

  # Crucial for S6 v3 execution from /run (Fixes the Permission Denied on /run/s6/...)
  /run/ r,
  /run/** rwkix,

  # Addon specific scripts and config
  /run.sh rix,
  /etc/nginx/** r,
  /tmp/ r,
  /tmp/** rwk,
  /tmp/nginx.conf r,
  /var/lib/nginx/ r,
  /var/lib/nginx/** rwk,

  # Logging and System
  /dev/stdout rw,
  /dev/stderr rw,
  /dev/tty rw,
  /dev/null rw,

  # Bashio and Options
  /usr/lib/bashio/** r,
  /data/options.json r,

  # Signals for process management
  signal (send, receive) peer=frigate_ingress_proxy,
  signal (receive) peer=unconfined,
  ptrace (trace) peer=unconfined,
}

#include <tunables/global>

# Profile name MUST match the slug in config.yaml
profile frigate_ingress_proxy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities required for S6-Overlay and process management
  capability setgid,
  capability setuid,
  capability dac_override,
  capability dac_read_search,
  capability net_bind_service,
  capability chown,
  capability fowner,
  capability kill,

  # Network access for the proxy
  network inet stream,
  network inet6 stream,

  # S6-Overlay and System core execution
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /sbin/nginx rix,
  /usr/sbin/nginx rix,
  /package/** rix,
  /command/** rix,

  # Allow execution of your startup script (CRITICAL: added rix)
  /run.sh rix,

  # Writable locations for S6 runtime and Nginx (CRITICAL: broadened /run)
  /run/ r,
  /run/** rwk,
  
  # Temporary and Config files
  /tmp/ r,
  /tmp/** rwk,
  /etc/nginx/** r,

  # Nginx internal fallback paths (Fixes the Permission Denied on error.log)
  /var/lib/nginx/ r,
  /var/lib/nginx/** rwk,
  /var/log/nginx/ r,
  /var/log/nginx/** rwk,

  # Logging and System
  /dev/stdout rw,
  /dev/stderr rw,
  /dev/tty rw,
  /dev/null rw,

  # Bashio and Home Assistant infrastructure
  /usr/lib/bashio/** r,
  /data/options.json r,

  # Process management signals
  signal (send, receive) peer=frigate_ingress_proxy,
  signal (receive) peer=unconfined,
  ptrace (trace) peer=unconfined,
}

#include <tunables/global>

# Profile name MUST match the add-on slug: frigate_ingress_proxy
profile frigate_ingress_proxy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities required for process management and Nginx
  capability setgid,
  capability setuid,
  capability dac_override,
  capability net_bind_service,

  # Network access for proxying
  network inet stream,
  network inet6 stream,

  # S6-Overlay core execution (Fixes the /init and s6-overlay-suexec errors)
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /sbin/nginx rix,
  /usr/sbin/nginx rix,
  /package/** rix,
  /command/** rix,

  # Allow execution of the startup script
  /run.sh r,

  # Runtime directories and process IDs
  /run/s6/** rwk,
  /run/s6-rc*/** rwk,
  /run/service/** rwk,
  /run/nginx.pid rwk,
  /tmp/nginx.pid rwk,

  # Nginx configuration and temporary file access
  /etc/nginx/** r,
  /tmp/nginx.conf r,
  /tmp/nginx/ r,
  /tmp/nginx/** rwk,
  
  # Nginx system paths (Fixes the Permission Denied on error.log)
  /var/lib/nginx/ r,
  /var/lib/nginx/** rwk,
  /var/log/nginx/ r,
  /var/log/nginx/** rwk,

  # System and logging access
  /dev/stdout rw,
  /dev/stderr rw,
  /dev/tty rw,
  /dev/null rw,

  # Bashio and Home Assistant options access
  /usr/lib/bashio/** r,
  /data/options.json r,

  # Allow the container to manage its own signals and processes
  signal (receive) peer=unconfined,
  ptrace (trace) peer=unconfined,
}

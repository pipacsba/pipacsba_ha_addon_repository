#include <tunables/global>

profile %%SLUG%% flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities - bashio and s6 need these to manage the container
  capability fowner,
  capability chown,
  capability dac_override,
  capability dac_read_search,
  capability setuid,
  capability setgid,
  capability net_bind_service,
  capability kill,

  # Signals
  signal (receive) peer=unconfined,
  signal (send,receive) peer=%%SLUG%%,
  signal (send) set=(kill,term,int,hup,cont),

  # Core System Paths (Added sbin, local/bin, and mmap support)
  /init rixmk,
  /bin/** rixmk,
  /sbin/** rixmk,
  /usr/bin/** rixmk,
  /usr/sbin/** rixmk,
  /usr/local/bin/** rixmk,
  /lib/** rmixk,
  /usr/lib/** rmixk,
  /usr/local/lib/** rmixk,

  # s6-overlay v3 Specifics
  /command/** rixmk,
  /package/** rixmk,
  /etc/s6-overlay/** rix,
  /run/s6/** rwixk,
  /run/s6-rc*/** rwixk,
  /run/service/** rwixk,

  # Service Directories - Needed for s6 to launch your scripts
  /etc/services.d/** rwixk,
  /etc/cont-init.d/** rix,
  /etc/cont-finish.d/** rix,

  # Bashio Library (The scripts bashio actually runs)
  /usr/lib/bashio/** rix,

  # Runtime, Temp, and Networking
  /run/{,**} rwk,
  /tmp/** rwk,
  /dev/tty rw,
  /etc/terminfo/** r,
  /etc/inputrc r,
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,
  unix (receive, send) type=stream,
  unix (receive, send) type=dgram,

  # Application and Data
  /app/** rwixm,
  /data/** rw,
  /share/** rw,
  /config/** rw,
}

#include <tunables/global>

profile %%SLUG%% flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities - Needed for s6-overlay and bashio to manage the addon
  capability fowner,
  capability chown,
  capability dac_override,
  capability dac_read_search,
  capability setuid,
  capability setgid,
  capability net_bind_service,
  capability kill,

  # Signals
  signal (receive) peer=unconfined,
  signal (send,receive) peer=%%SLUG%%,
  signal (send) set=(kill,term,int,hup,cont),

  # Core System Paths - Added 'm' and 'k' for Python/Busybox stability
  /init rixmk,
  /bin/** rixmk,
  /sbin/** rixmk,
  /usr/bin/** rixmk,
  /usr/sbin/** rixmk,
  /usr/local/bin/** rixmk,
  /lib/** rmixk,
  /usr/lib/** rmixk,
  /usr/local/lib/** rmixk,

  # Bashio (The library scripts)
  # Needs 'r' to be read by bash and 'ix' to be executed
  /usr/lib/bashio/** rix,

  # s6-overlay v3 Requirements
  /command/** rixmk,
  /package/** rixmk,
  /etc/s6-overlay/** rix,
  /run/s6/** rwixk,
  /run/s6-rc*/** rwixk,
  /run/service/** rwixk,
  
  # Service management
  /etc/services.d/** rwixk,
  /etc/cont-init.d/** rix,
  /etc/cont-finish.d/** rix,

  # System Configs (Needed for DNS and Shell environment)
  /etc/** r,

  # Runtime and Temp
  /run/{,**} rwk,
  /tmp/** rwk,
  /dev/tty rw,

  # Application Paths
  /app/** rwixm,
  /data/** rw,
  /share/** rw,
  /config/** rw,

  # Networking (Crucial for DNS and Ingress)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,
  unix (receive, send) type=stream,
  unix (receive, send) type=dgram,
}

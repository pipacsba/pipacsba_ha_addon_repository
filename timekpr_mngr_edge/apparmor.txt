#include <tunables/global>

# Use %%SLUG%% so Home Assistant names the profile correctly
profile %%SLUG%% flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Instead of the global "file," we grant access to the specific 
  # system areas your Python app and s6-overlay need to "boot"
  /bin/** rix,
  /lib/** rmk,
  /usr/lib/** rmk,
  /usr/bin/** rix,
  /usr/local/lib/python3*/** rmk,
  /etc/** r,            # Needed to read service configs and resolve DNS
  /init rix,            # The s6-overlay entrypoint

  # Capabilities
  capability fowner,
  capability chown,
  capability dac_override,
  capability setuid,
  capability setgid,
  capability net_bind_service,

  # Signals
  signal (receive) peer=unconfined,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay & Runtime (Added 'w' and 'k' for locking)
  /run/s6/** rwixk,
  /run/s6-rc*/** rwixk,
  /run/service/** rwixk,
  /run/{,**} rwk,
  /package/** rix,
  /command/** rix,
  /etc/services.d/** rwixk,
  /etc/cont-init.d/** rix,
  /dev/tty rw,
  /tmp/** rwk,

  # Your Application Specific Paths
  /app/** rwix,
  /data/** rw,
  /share/** rw,
  /config/** rw,

  # Networking
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,
}
